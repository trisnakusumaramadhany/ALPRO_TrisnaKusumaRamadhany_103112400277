program AnalisisProfit
variabel
    bulan1, bulan2, profit: real
algoritma
    input(bulan1, bulan2)
    profit = bulan2 - bulan1
    if profit > 0 then
        output("Peningkatan Sebesar", profit)
    else if profit == 0 then
        output("Tetap")
    else
        output("Penurunan Sebesar", -profit)
    end if
end program
