program HitungJumlahUang
variabel
    uang, jumlah: integer
    selesai: boolean
algoritma
    jumlah = 0
    selesai = false
    while selesai = false do
        input(uang)
        jumlah = jumlah + uang
        if uang <= 0 then
            selesai = true
        end if
    end while
    output(jumlah)
end program
